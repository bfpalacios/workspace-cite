CREATE PROCEDURE [dbo].[EliminarUsuariocite]
            @id int
        AS
            BEGIN
	
                DELETE FROM dbo.USUARIO_ROL
                WHERE IDUSUARIO = @id

                WHERE IDUSUARIO = @id
                
                
END
